# [Game DB]

## Mail Table

``` sql
DROP DATABASE IF EXISTS game_db;
CREATE DATABASE IF NOT EXISTS game_db;

USE game_db;

DROP TABLE IF EXISTS game_db.`mail`;
CREATE TABLE IF NOT EXISTS game_db.`mail`
(
    code BIGINT UNSIGNED AUTO_INCREMENT NOT NULL PRIMARY KEY COMMENT '우편 번호',
    player_id BIGINT NOT NULL COMMENT '고유 번호',
    title VARCHAR(50) NOT NULL COMMENT '우편 제목',
    content VARCHAR(100) NOT NULL COMMENT '우편 내용',
    exp INT NOT NULL COMMENT '우편 유효 기간',
    type INT NOT NULL COMMENT '우편 종류',
    state_type INT NOT NULL COMMENT '우편 상태',
    item_code INT NOT NULL COMMENT '아이템 코드',
    count INT NOT NULL COMMENT '지급 개수',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '생성 날짜'
) COMMENT '우편함의 정보';
```sql
